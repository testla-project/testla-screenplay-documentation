import CodeBlock from '@theme/CodeBlock';
import txtReport from './example-reports/txt-report.js';
import jsonReport from './example-reports/json-report.js';

export default function ExampleTextReportOutput({ type = 'txt' }) {
  const reportOutput = type === 'txt' ? txtReport : JSON.stringify(jsonReport, null, 2);

  return (
    <div style={{ maxHeight: '400px', overflowY: 'auto' }}>
      <CodeBlock language={type === 'txt' ? 'text' : 'json'}>{reportOutput}</CodeBlock>
    </div>
  );
}
