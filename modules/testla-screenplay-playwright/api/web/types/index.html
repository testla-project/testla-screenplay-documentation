Types

Selector​

type Selector = string | Locator | LazySelector;

LazySelector​

type LazySelector = (page: Page | FrameLocator) => Locator;

SelectorOptions​

type SelectorOptions = {
    hasText?: string | RegExp | undefined;
    subSelector?: SubSelector;
    timeout?: number;
    state?: SelectorOptionsState;
    evaluateVisible?: boolean;
};

SubSelector​

type SubSelector = [
    Selector, SelectorOptions?
];

SelectorOptionsState​

type SelectorOptionsState = 'attached' | 'detached' | 'visible' | 'hidden';

ClickOptions​

type ClickOptions = {
    button?: 'left'|'right'|'middle';
    clickCount?: number;
    delay?: number;
    force?: boolean;
    modifiers?: Array<'Alt'|'Control'|'Meta'|'Shift'>;
    noWaitAfter?: boolean;
    position?: {
        x: number;
        y: number;
    };
    trial?: boolean;
};

DoubleClickOptions​

type DoubleClickOptions = {
    button?: 'left'|'right'|'middle';
    delay?: number;
    force?: boolean;
    modifiers?: Array<'Alt'|'Control'|'Meta'|'Shift'>;
    noWaitAfter?: boolean;
    position?: {
        x: number;
        y: number;
    };
    trial?: boolean;
};

Maskable​

type Maskable = {
    maskInLogs?: boolean;
};
